function add_metabot_widget() {
    wp_enqueue_script(
        'metabot-widget',
        'https://metabot-ai-frontend-production.up.railway.app/metabot-widget-standalone.js',
        array(),
        '1.0.0',
        true
    );
}
add_action('wp_enqueue_scripts', 'add_metabot_widget');

‚öõÔ∏è React Integration

Use with React applications:

import { useEffect } from 'react';

function App() {
    useEffect(() => {
        const script = document.createElement('script');
        script.src = 'https://metabot-ai-frontend-production.up.railway.app/metabot-widget-standalone.js';
        script.async = true;
        document.body.appendChild(script);
        
        return () => {
            document.body.removeChild(script);
            window.Metabot?.destroy();
        };
    }, []);

    return <div>Your React App</div>;
}
